Long running busybox pod

apiVersion: v1
kind: Pod
metadata:
  labels:
    app: debug-pod
  name: debug-pod
spec:
  containers:
  - args:
    - while true; do sleep 10000; done
    command:
    - sh
    - -c
    image: busybox
    imagePullPolicy: Always
    name: debug-pod
    resources:
      requests:
        cpu: 200m
        memory: 100Mi
      limits:
        cpu: 500m
        memory: 200Mi
